1    package org.example.bankwithspringboot.controller;

3    import lombok.Getter;
4    import org.example.bankwithspringboot.model.User;
5    import org.example.bankwithspringboot.service.UserService;
6    import org.springframework.http.ResponseEntity;
7    import org.springframework.web.bind.annotation.*;
8    
9    import java.util.Optional;
10   
11   @RestController
12   @RequestMapping("/api/users")
13   @Getter
14   public class UserController {
15   
16       private final UserService service;
17   
18       public UserController(UserService service){
19           this.service = service;
20       }
21   
22       @PostMapping("/register")
23       public ResponseEntity<User> registerUser(@RequestBody User user){
24           User savedUser = service.registeruser(user);
25           return ResponseEntity.ok(savedUser);
26       }
27   
28       @PostMapping("/login")
29       public ResponseEntity<String> loginUser(@RequestBody User user){
30           Optional<User> loggedInUser = service.loginUser(user.getEmail(),user.getPassword());
31           return loggedInUser.map(value -> ResponseEntity.ok("Login successful! Welcome " + value.getName())).orElseGet(() -> ResponseEntity.status(401).body("Invalid email or password"));
32       }
33   }
